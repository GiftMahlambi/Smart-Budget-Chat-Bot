<script>
    // Generate stars
    const starsEl = document.getElementById('stars-container');
    for (let i = 0; i < 120; i++) {
        const s = document.createElement('div');
        s.className = 'star';
        const size = Math.random() * 2.5 + 0.5;
        s.style.cssText = `
            width:${size}px; height:${size}px;
            top:${Math.random() * 100}%;
            left:${Math.random() * 100}%;
            opacity:${Math.random() * 0.7 + 0.1};
            animation-duration:${Math.random() * 4 + 2}s;
            animation-delay:${Math.random() * 5}s;
        `;
        starsEl.appendChild(s);
    }

    // Enter key to send
    document.getElementById('userInput').addEventListener('keydown', function(e) {
        if (e.key === 'Enter') sendMessage();
    });

    function addMsg(text, isUser) {
        const box = document.getElementById('chatbox');
        const wrap = document.createElement('div');
        wrap.className = 'msg-wrap ' + (isUser ? 'user' : 'bot');
        wrap.innerHTML = `
            <div class="bubble ${isUser ? 'user' : 'bot'}">${text}</div>
            <div class="msg-label">${isUser ? 'You' : 'Budget Bot'}</div>
        `;
        box.appendChild(wrap);
        box.scrollTop = box.scrollHeight;
    }

    function addTyping() {
        const box = document.getElementById('chatbox');
        const t = document.createElement('div');
        t.className = 'msg-wrap bot';
        t.id = 'typing';
        t.innerHTML = `
            <div class="bubble bot">
                <span class="typing-dot"></span>
                <span class="typing-dot"></span>
                <span class="typing-dot"></span>
            </div>
        `;
        box.appendChild(t);
        box.scrollTop = box.scrollHeight;
    }

    function removeTyping() {
        const t = document.getElementById('typing');
        if (t) t.remove();
    }

    function sendMessage() {
        const inp = document.getElementById('userInput');
        const message = inp.value.trim();
        if (!message) return;
        inp.value = '';
        addMsg(message, true);
        addTyping();

        fetch('/chat', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ message })
        })
        .then(res => res.json())
        .then(data => {
            
            removeTyping();
            addMsg(data.reply, false);
        })
        .catch(() => {
            removeTyping();
            addMsg("Sorry, I couldn't reach the server. Please try again!", false);
        });
    }

    function quickSend(text) {
        document.getElementById('userInput').value = text;
        sendMessage();
    }
</script>
